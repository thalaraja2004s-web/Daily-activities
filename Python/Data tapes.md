"""
Python Data Types Examples
A data type tells Python what kind of data is stored in a variable.
"""

# 1. Integer - Stores whole numbers
a = 10
b = 50
print("Integer value:", a)
print("Type:", type(a))


# 2. Float - Stores decimal numbers
a = 3.5
print("\nFloat value:", a)
print("Type:", type(a))


# 3. String - Stores text or characters
a = "Raja"
print("\nString value:", a)
print("Type:", type(a))


# 4. List - Stores multiple values using [ ]
a = [10, 20, 30]
print("\nList:", a)
print("Type:", type(a))


# 5. Tuple - Stores multiple values using ( ) but cannot be changed
a = (1, 2, 3)
print("\nTuple:", a)
print("Type:", type(a))


# 6. Set - Stores unique values using { }
a = {1, 2, 3, 3}
print("\nSet:", a)
print("Type:", type(a))


# 7. Dictionary - Stores data in key : value pairs
a = {"name": "Raja", "age": 21}
print("\nDictionary:", a)
print("Type:", type(a))
